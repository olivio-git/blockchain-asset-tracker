# Crear un nuevo repositorio
gh repo create nombre-del-repositorio --public

# Crear un repositorio privado
gh repo create nombre-del-repositorio --private

# Clonar un repositorio
gh repo clone usuario/repositorio

# Listar tus repositorios
gh repo list

# Ver detalles de un repositorio
gh repo view nombre-del-repositorio

# Crear un fork de un repositorio
gh repo fork usuario/repositorio
